# Minimap Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a persistent minimap to the game UI that shows district zones and player position, with click-to-expand resident lists and fade-transition teleportation.

**Architecture:** A React overlay component renders a `<canvas>` element that displays a cached minimap texture generated once by Phaser's RenderTexture. District zones are React `<div>` overlays on the canvas. Clicking a zone opens a resident list panel; clicking a resident triggers a fade-out/teleport/fade-in sequence in Phaser via PhaserBridge events.

**Tech Stack:** Phaser 3 (RenderTexture, Camera, Tweens), React 18 (hooks, canvas ref), TypeScript, Zustand (gameStore), PhaserBridge (event bus)

```ts
  private generateMinimapTexture(): void {
    const map = this.make.tilemap({ key: 'village' })
    const MINIMAP_W = 180
    const MINIMAP_H = 130
    const scaleX = MINIMAP_W / map.widthInPixels
    const scaleY = MINIMAP_H / map.heightInPixels

    const rt = this.add.renderTexture(0, 0, MINIMAP_W, MINIMAP_H)
    rt.setVisible(false)

    // Draw each visible tilemap layer scaled down
    const layerNames = [
      'Bottom Ground', 'Exterior Ground', 'Exterior Decoration L1',
      'Exterior Decoration L2', 'Interior Ground', 'Interior Furniture L1',
      'Interior Furniture L2', 'Wall',
    ]

    // Use a temporary camera to render at scale
    const tempCam = this.cameras.add(0, 0, MINIMAP_W, MINIMAP_H, false)
    tempCam.setZoom(scaleX) // scaleX ≈ 0.04, renders entire map into 180px
    tempCam.setScroll(0, 0)

    // Snapshot the main scene layers into the render texture
    // Alternative approach: draw each layer's tilemap into RT
    // Since RenderTexture.draw doesn't support camera zoom easily,
    // we'll snapshot the canvas directly
    this.game.renderer.snapshotArea(
      0, 0, map.widthInPixels * scaleX, map.heightInPixels * scaleY,
      (image) => {
        const img = image as HTMLImageElement
        const dataUrl = img.src || this.textures.createCanvas('minimap_snap', MINIMAP_W, MINIMAP_H)?.canvas.toDataURL() || ''

        // If snapshotArea gave us a usable image
        if (img.src) {
          useGameStore.getState().setMinimapTexture(img.src)
        }
        bridge.emit('minimap:texture', { dataUrl: img.src || dataUrl })
      }
    )

    this.cameras.remove(tempCam)
    rt.destroy()
  }
```

**Note:** Phaser's `snapshotArea` is async and callback-based. The approach above captures the visible rendered area. However, since the main camera only shows a portion of the map, a more reliable approach is to temporarily set the main camera to show the full map, snapshot, then restore. Replace the method body with this improved version:

```ts
  private generateMinimapTexture(): void {
    const cam = this.cameras.main
    const origScrollX = cam.scrollX
    const origScrollY = cam.scrollY
    const origZoom = cam.zoom

    // Temporarily zoom out to capture entire map
    const mapW = cam.getBounds().width
    const mapH = cam.getBounds().height
    const MINIMAP_W = 180
    const MINIMAP_H = 130
    const zoom = Math.min(cam.width / mapW, cam.height / mapH)

    cam.setZoom(zoom)
    cam.setScroll(0, 0)
    cam.stopFollow()

    // Wait one frame for the camera to render, then snapshot
    this.time.delayedCall(50, () => {
      this.game.renderer.snapshotArea(
        0, 0,
        Math.ceil(mapW * zoom),
        Math.ceil(mapH * zoom),
        (image) => {
          // Restore camera
          cam.setZoom(origZoom)
          cam.setScroll(origScrollX, origScrollY)
          if (this.player) {
            cam.startFollow(this.player, true, 0.1, 0.1)
          }

          const img = image as HTMLImageElement
          if (img.src) {
            useGameStore.getState().setMinimapTexture(img.src)
          }
        }
      )
    })
  }
```

```ts
  private teleportTo(tileX: number, tileY: number): void {
    if (this.isTeleporting) return

    const cam = this.cameras.main
    const targetX = tileX * TILE_SIZE + TILE_SIZE / 2
    const targetY = tileY * TILE_SIZE + TILE_SIZE

    // Phase 1: Fade out (300ms)

      cam.fadeIn(500, 0, 0, 0)

        this.isTeleporting = false
        bridge.emit('teleport:complete', { tileX, tileY })
      })
    })
  }
```

```tsx
import { useEffect, useRef } from 'react'
import { useGameStore } from '../../stores/gameStore'

const MAP_TILES_W = 140
const MAP_TILES_H = 100
const MINIMAP_W = 180
const MINIMAP_H = 130

export function MinimapCanvas() {
  const canvasRef = useRef<HTMLCanvasElement>(null)
  const imgRef = useRef<HTMLImageElement | null>(null)
  const textureUrl = useGameStore((s) => s.minimapTextureUrl)
  const playerTileX = useGameStore((s) => s.playerTileX)
  const playerTileY = useGameStore((s) => s.playerTileY)
  const viewport = useGameStore((s) => s.cameraViewport)

  // Load texture image when URL changes
  useEffect(() => {
    if (!textureUrl) return
    const img = new Image()
    img.onload = () => { imgRef.current = img }
    img.src = textureUrl
  }, [textureUrl])

  // Redraw canvas on every relevant state change
  useEffect(() => {
    const canvas = canvasRef.current
    if (!canvas) return
    const ctx = canvas.getContext('2d')
    if (!ctx) return

    ctx.clearRect(0, 0, MINIMAP_W, MINIMAP_H)

    // Draw cached map texture
    if (imgRef.current) {
      ctx.drawImage(imgRef.current, 0, 0, MINIMAP_W, MINIMAP_H)
    } else {
      // Fallback: dark green background
      ctx.fillStyle = '#2d5016'
      ctx.fillRect(0, 0, MINIMAP_W, MINIMAP_H)
    }

    // Draw camera viewport rectangle
    if (viewport) {
      const vx = (viewport.x / MAP_TILES_W) * MINIMAP_W
      const vy = (viewport.y / MAP_TILES_H) * MINIMAP_H
      const vw = (viewport.w / MAP_TILES_W) * MINIMAP_W
      const vh = (viewport.h / MAP_TILES_H) * MINIMAP_H
      ctx.strokeStyle = 'rgba(255,255,255,0.5)'
      ctx.lineWidth = 1
      ctx.strokeRect(vx, vy, vw, vh)
    }

    // Draw player dot
    const px = (playerTileX / MAP_TILES_W) * MINIMAP_W
    const py = (playerTileY / MAP_TILES_H) * MINIMAP_H
    ctx.beginPath()
    ctx.arc(px, py, 4, 0, Math.PI * 2)
    ctx.fillStyle = '#ffffff'
    ctx.fill()
    ctx.strokeStyle = '#3b82f6'
    ctx.lineWidth = 2
    ctx.stroke()
  }, [playerTileX, playerTileY, viewport, textureUrl])

  return (
    <canvas
      ref={canvasRef}
      width={MINIMAP_W}
      height={MINIMAP_H}
      style={{ display: 'block', borderRadius: 6 }}
    />
  )
}
```

```tsx
import { useEffect, useState } from 'react'

import type { ResidentData } from '../../game/GameScene'

import { DISTRICTS, type DistrictKey } from './DistrictZones'

interface Props {
  district: DistrictKey
  onClose: () => void
}

export function ResidentPanel({ district, onClose }: Props) {
  const [residents, setResidents] = useState<ResidentData[]>([])
  const config = DISTRICTS.find((d) => d.key === district)!

  useEffect(() => {
    fetch('/residents')
      .then((r) => r.json())
      .then((all: ResidentData[]) => {
        setResidents(all.filter((r) => {
          // Match resident to district by checking if their tile is within the district bounds
          const rect = config.tileRect
          return r.tile_x >= rect.x && r.tile_x <= rect.x + rect.w
            && r.tile_y >= rect.y && r.tile_y <= rect.y + rect.h
        }))
      })
  }, [district, config.tileRect])

  const handleTeleport = (r: ResidentData) => {
    // Teleport to 1 tile left of the resident to avoid overlap
    bridge.emit('minimap:teleport', {
      tileX: r.tile_x - 1,
      tileY: r.tile_y,
      residentSlug: r.slug,

    onClose()

  return (
    <div style={{
      position: 'absolute',
      top: 0,
      left: 188, // 180px minimap + 8px gap
```
